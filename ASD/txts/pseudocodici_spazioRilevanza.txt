//nota: uno stato deve contenere la decorazione incontrata fino ad esso
creaSpazioRilevanza(mappaStatiTraiettorie, rete, stato, traiettoria){
    if(esisteTransizioneDisponibile(rete, statoIniziale))
    then
        ultimaTransizioneEseguita <- eseguiTransizioneDisponibile(rete)
        nuovoStato <- calcolaStato(rete)
        add(traiettoria, ultimaTransizioneEseguita)
        if(mappaStatiTraiettorie !contains nuovoStato)
        then
            add(mappaStatiTraiettorie, nuovoStato, traiettoria)
        else
            addTraiettoria(mappaStatiTraiettorie, trovaStatoCoincidente(nuovoStato), traiettoria)
        endif
        return creaSpazioRilevanza(mappaStatiTraiettorie, rete, statoAttuale, traiettoria)
    else return
}
